class Desenvolvedor:
    def __init__(self, nome, cidade, idade, hobbies):
        self.nome = nome
        self.cidade = cidade
        self.idade = idade
        self.hobbies = hobbies

    def __str__(self):
        return (
          f"Nome: {self.nome}\n"
          f"Mora em {self.cidade}\n"
          f"Tem {self.idade} anos\n"
          f"Hobbies: {', '.join(self.hobbies)}.\n"
        )


eu = Desenvolvedor(
    "Matheus",
    "Nova Friburgo - RJ",

)
